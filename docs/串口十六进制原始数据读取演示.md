```cpp
    static QString bytesToHexSpaced(const QByteArray &data);

    static QString bytesToHexDump(const QByteArray &data, int bytesPerLine = 16);

    void onReadyRead();

    QSerialPort *m_port = nullptr;
    QByteArray m_rxAccum; // 可选：若你想连续观察原始流
```

## 2. 实现文件片段

```cpp
#include "SerialHexDemo.h" // 按你实际文件名调整
#include <QDebug>

SerialHexDemo::SerialHexDemo(QObject *parent)
    : QObject(parent)
    , m_port(new QSerialPort(this))
{
    connect(m_port, &QSerialPort::readyRead, this, &SerialHexDemo::onReadyRead);
    connect(m_port, &QSerialPort::errorOccurred, this, [this](QSerialPort::SerialPortError e) {
        if (e == QSerialPort::NoError)
            return;
        qWarning() << "Serial error:" << m_port->errorString();
    });
}

SerialHexDemo::~SerialHexDemo()
{
    closePort();
}

bool SerialHexDemo::openPort(const QString &portName, qint32 baudRate)
{
    closePort();
    m_port->setPortName(portName);
    m_port->setBaudRate(baudRate);
    m_port->setDataBits(QSerialPort::Data8);
    m_port->setParity(QSerialPort::NoParity);
    m_port->setStopBits(QSerialPort::OneStop);
    m_port->setFlowControl(QSerialPort::NoFlowControl);
    // 可选：增大驱动侧缓冲，减少 read 次数（非必须）
    m_port->setReadBufferSize(32 * 1024);

    if (!m_port->open(QIODevice::ReadOnly)) {
        qWarning() << "Open failed:" << m_port->errorString();
        return false;
    }
    return true;
}

void SerialHexDemo::closePort()
{
    if (m_port->isOpen())
        m_port->close();
}

QString SerialHexDemo::bytesToHexSpaced(const QByteArray &data)
{
    if (data.isEmpty())
        return QString();
    // Qt6：separator 用 char，常见为空格
    return QString::fromLatin1(data.toHex(' '));
}

QString SerialHexDemo::bytesToHexDump(const QByteArray &data, int bytesPerLine)
{
    if (bytesPerLine <= 0)
        bytesPerLine = 16;

    QString out;
    out.reserve(data.size() * 3 + data.size() / bytesPerLine * 8);

    for (int offset = 0; offset < data.size(); offset += bytesPerLine) {
        const QByteArray line = data.sliced(offset, qMin(bytesPerLine, data.size() - offset));
        out += QStringLiteral("%1  %2\n")
                   .arg(offset, 6, 16, QChar(u'0'))
                   .arg(QString::fromLatin1(line.toHex(' ')));
    }
    return out;
}

void SerialHexDemo::onReadyRead()
{
    const QByteArray chunk = m_port->readAll();
    if (chunk.isEmpty())
        return;

    // 方式 A：本次 read 的原始十六进制（最常用）
    qDebug().noquote() << "RX chunk (hex):" << bytesToHexSpaced(chunk);

    // 方式 B：带偏移的多行 hexdump（数据稍长时更清晰）

}
```

## 使用说明（要点）

1. `readyRead` 里 **`readAll()` 返回的字节数是任意的**，同一次回调可能是半包、一包或多包粘在一起；十六进制打印用于**确认线上原始字节**，与后续「组帧解析」是两步。  
2. 高频数据不要用 `qDebug` 无节流打印，会拖慢事件循环；调试时可加计数器每 N 次打印一行。  
3. `QByteArray::toHex(' ')` 得到的是 **小写** 十六进制 ASCII；若要大写可对 `QString` 再 `toUpper()`。
