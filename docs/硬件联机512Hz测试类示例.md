# 硬件联机测试类示例（验证 1 秒约 512 个 RAW 点）

> 目标：测试 `thinkgear` 各层串起来后，是否稳定解析出 **1 秒约 512 个 RAW 数据点**。  
> 本文只给代码示例，方便你复制到工程中测试；不直接改现有源码。

---

## 一、测试思路

链路连接如下：

`SerialPort -> ThinkGearFrameAssembler -> ThinkGearPayloadParser -> RawToUvProcessor`

测试类每秒统计：
- 本秒 `rawReceived` 次数（raw 点数）
- 本秒组帧成功数
- 解析告警数
- 与目标 `512` 的偏差

判定规则（可调）：
- `rawCountPerSec` 在 `512 ± 25` 认为通过（约 5% 容差）

---

## 二、测试类头文件示例：`thinkgear/thinkgearlinktester.h`

```cpp
#ifndef THINKGEARLINKTESTER_H
#define THINKGEARLINKTESTER_H

#include <QObject>
#include <QTimer>
#include <QtGlobal>

class SerialPort;
class ThinkGearFrameAssembler;
class ThinkGearPayloadParser;
class RawToUvProcessor;

class ThinkGearLinkTester : public QObject
{
    Q_OBJECT

public:
    explicit ThinkGearLinkTester(QObject *parent = nullptr);
    ~ThinkGearLinkTester() override;

    // 启动测试：指定串口名（例如 "COM7"）
    bool start(const QString &portName, qint32 baudRate = 57600);
    void stop();

    void setTargetRawPerSec(int v) { m_targetRawPerSec = v; } // 默认 512
    void setTolerance(int v) { m_tolerance = v; }             // 默认 ±25

signals:
    // 每秒输出一次统计（可连到 UI 或日志）
    void secondReport(quint64 secIndex,
                      int rawCountPerSec,
                      int frameCountPerSec,
                      int warningCountPerSec,
                      bool pass);

    void testMessage(const QString &msg);

private slots:
    void onRawUvReady(double uv);
    void onFrameReady(const QByteArray &frame);
    void onParseWarning(const QString &msg);
    void onPortError(const QString &msg);
    void onTick1s();

private:
    SerialPort *m_serial = nullptr;
    ThinkGearFrameAssembler *m_assembler = nullptr;
    ThinkGearPayloadParser *m_parser = nullptr;
    RawToUvProcessor *m_converter = nullptr;

    QTimer m_tick;

    // 每秒计数器
    int m_rawCountSec = 0;
    int m_frameCountSec = 0;
    int m_warningCountSec = 0;
    quint64 m_secIndex = 0;

    // 阈值
    int m_targetRawPerSec = 512;
    int m_tolerance = 25;
};

#endif // THINKGEARLINKTESTER_H
```

## 三、测试类实现示例：`thinkgear/thinkgearlinktester.cpp`

```cpp
#include "thinkgearlinktester.h"

#include "device/serialport.h"
#include "thinkgear/thinkgearframeassembler.h"
#include "thinkgear/thinkgearpayloadparser.h"
#include "thinkgear/rawtouvprocessor.h"

#include <QDebug>

ThinkGearLinkTester::ThinkGearLinkTester(QObject *parent)
    : QObject(parent)
{
    m_serial = new SerialPort(this);
    m_assembler = new ThinkGearFrameAssembler(this);
    m_parser = new ThinkGearPayloadParser(this);
    m_converter = new RawToUvProcessor(this);

    // 链路连接
    connect(m_serial, &SerialPort::dataReceived,
            m_assembler, &ThinkGearFrameAssembler::onBytes);

    connect(m_assembler, &ThinkGearFrameAssembler::frameReady,
            this, &ThinkGearLinkTester::onFrameReady);
    connect(m_assembler, &ThinkGearFrameAssembler::frameReady,
            m_parser, &ThinkGearPayloadParser::parseFrame);

    connect(m_parser, &ThinkGearPayloadParser::rawReceived,
            m_converter, &RawToUvProcessor::onRaw);

    connect(m_converter, &RawToUvProcessor::uvValueReady,
            this, &ThinkGearLinkTester::onRawUvReady);

    connect(m_parser, &ThinkGearPayloadParser::parseWarning,
            this, &ThinkGearLinkTester::onParseWarning);

    connect(m_serial, &SerialPort::portError,
            this, &ThinkGearLinkTester::onPortError);

    // 1 秒统计定时器
    m_tick.setInterval(1000);
    connect(&m_tick, &QTimer::timeout, this, &ThinkGearLinkTester::onTick1s);
}

ThinkGearLinkTester::~ThinkGearLinkTester()
{
    stop();
}

bool ThinkGearLinkTester::start(const QString &portName, qint32 baudRate)
{
    stop();

    SerialPortConfig cfg;
    cfg.portName = portName;
    cfg.baudRate = baudRate;
    cfg.readBufferSize = 32 * 1024;
    m_serial->setConfig(cfg);

    const bool ok = m_serial->openReadOnly();
    if (!ok) {
        emit testMessage(QStringLiteral("串口打开失败"));
        return false;
    }

    m_rawCountSec = 0;
    m_frameCountSec = 0;
    m_warningCountSec = 0;
    m_secIndex = 0;
    m_tick.start();

    emit testMessage(QStringLiteral("测试开始：等待每秒统计..."));
    return true;
}

void ThinkGearLinkTester::stop()
{
    if (m_tick.isActive())
        m_tick.stop();
    if (m_serial && m_serial->isOpen())
        m_serial->close();
}

void ThinkGearLinkTester::onRawUvReady(double uv)
{
    Q_UNUSED(uv);
    ++m_rawCountSec;
}

void ThinkGearLinkTester::onFrameReady(const QByteArray &frame)
{
    Q_UNUSED(frame);
    ++m_frameCountSec;
}

void ThinkGearLinkTester::onParseWarning(const QString &msg)
{
    Q_UNUSED(msg);
    ++m_warningCountSec;
}

void ThinkGearLinkTester::onPortError(const QString &msg)
{
    emit testMessage(QStringLiteral("串口错误: %1").arg(msg));
}

void ThinkGearLinkTester::onTick1s()
{
    ++m_secIndex;

    const int low = m_targetRawPerSec - m_tolerance;
    const int high = m_targetRawPerSec + m_tolerance;
    const bool pass = (m_rawCountSec >= low && m_rawCountSec <= high);

    emit secondReport(m_secIndex, m_rawCountSec, m_frameCountSec, m_warningCountSec, pass);

    // 控制台打印便于快速观察
    qDebug().noquote()
        << QString("sec=%1 raw/s=%2 frame/s=%3 warn/s=%4 pass=%5")
               .arg(m_secIndex)
               .arg(m_rawCountSec)
               .arg(m_frameCountSec)
               .arg(m_warningCountSec)
               .arg(pass ? "YES" : "NO");

    // 清零进入下一秒
    m_rawCountSec = 0;
    m_frameCountSec = 0;
    m_warningCountSec = 0;
}
```

## 四、在 `MainWindow` 里最小调用示例

```cpp
#include "thinkgear/thinkgearlinktester.h"

// 成员
ThinkGearLinkTester *m_linkTester = nullptr;

// 例如构造函数里
m_linkTester = new ThinkGearLinkTester(this);

connect(m_linkTester, &ThinkGearLinkTester::secondReport, this,
        [](quint64 secIndex, int rawPerSec, int framePerSec, int warnPerSec, bool pass) {
            qDebug() << "[REPORT]"
                     << "sec" << secIndex
                     << "raw/s" << rawPerSec
                     << "frame/s" << framePerSec
                     << "warn/s" << warnPerSec
                     << "pass" << pass;
        });

connect(m_linkTester, &ThinkGearLinkTester::testMessage, this,
        [](const QString &m) { qDebug() << m; });

// 启动测试
m_linkTester->setTargetRawPerSec(512);
m_linkTester->setTolerance(25);
m_linkTester->start("COM7", 57600);
```

## 六、测试通过标准建议

连续 30 秒满足：
- `raw/s` 大部分秒数在 `512 ± 25`
- `warn/s` 持续接近 0（偶发可接受）
- 不出现串口错误与明显队列积压

如果 `raw/s` 明显偏低：
1. 先看串口波特率和端口是否正确  
2. 再看组帧/解析告警是否偏高  
3. 关闭大量 `qDebug`（日志过多会拖慢）  
4. 确认写盘线程与 UI 刷新不会阻塞采集线程
